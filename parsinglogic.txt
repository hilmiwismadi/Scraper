  saveToLocal(sessionId, data) {
    const outputDir = path.join(process.cwd(), 'output');  // <- output folder
    if (!fs.existsSync(outputDir)) {
      fs.mkdirSync(outputDir, { recursive: true });
    }

    const filename = `scraped-${sessionId.substring(0, 8)}-${Date.now()}.json`;
    const filepath = path.join(outputDir, filename);  // <- output/filename.json
    fs.writeFileSync(filepath, JSON.stringify(data, null, 2));
    return filepath;
  }
